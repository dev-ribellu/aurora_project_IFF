// SCORE CALCULATOR
// Auto-compute total CM from all missions marked status:"done"
function computeTotalCM() {
  let total = 0;
  [...MISSIONS_SOCLE, 
   ...TIMELINE_EVENTS.flatMap(e => e.missions || [])]
    .filter(m => m.status === "done")
    .forEach(m => total += m.points);
  return total;
}
